; ModuleID = 'moudle'
source_filename = "moudle"

define i32 @main() {
mainEntry:
  %a = alloca i32, align 4
  store i32 0, i32* %a, align 4
  %count = alloca i32, align 4
  store i32 0, i32* %count, align 4
  br label %condition

condition:                                        ; preds = %the_next, %mainEntry
  %a1 = load i32, i32* %a, align 4
  %le = icmp sle i32 %a1, 0
  %cond7 = zext i1 %le to i32
  %cond8 = icmp ne i32 0, %cond7
  br i1 %cond8, label %while_body, label %next_block

while_body:                                       ; preds = %condition
  %a2 = load i32, i32* %a, align 4
  %subtmp = sub i32 %a2, 1
  store i32 %subtmp, i32* %a, align 4
  %count3 = load i32, i32* %count, align 4
  %addtmp = add i32 %count3, 1
  store i32 %addtmp, i32* %count, align 4
  br label %condition4

next_block:                                       ; preds = %condition, %the_true
  %count9 = load i32, i32* %count, align 4
  ret i32 %count9

condition4:                                       ; preds = %while_body
  %a5 = load i32, i32* %a, align 4
  %lt = icmp slt i32 %a5, -20
  %cmp = icmp eq i1 %lt, i32 0
  %a6 = load i32, i32* %a, align 4
  %gt = icmp sgt i32 %a6, -40
  %result = select i1 %cmp, i32 0, i1 %gt
  %cond = icmp ne i32 0, %result
  br i1 %cond, label %the_true, label %the_false

the_true:                                         ; preds = %condition4
  br label %next_block
  br label %the_next

the_false:                                        ; preds = %condition4

the_next:                                         ; preds = %the_true
  br label %condition
}
