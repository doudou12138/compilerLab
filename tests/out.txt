; ModuleID = 'moudle'
source_filename = "moudle"

@a = global i32 10

define i32 @main() {
mainEntry:
  br label %condition

condition:                                        ; preds = %while_body, %mainEntry
  %a = load i32, i32* @a, align 4
  %gt = icmp sgt i32 %a, 1
  %cond = zext i1 %gt to i32
  %cond2 = icmp ne i32 0, %cond
  br i1 %cond2, label %while_body, label %next_block

while_body:                                       ; preds = %condition
  %a1 = load i32, i32* @a, align 4
  %subtmp = sub i32 %a1, 1
  store i32 %subtmp, i32* @a, align 4
  br label %condition

next_block:                                       ; preds = %condition
  %a3 = load i32, i32* @a, align 4
  ret i32 %a3
}
