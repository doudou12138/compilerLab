source_filename = "moudle"

define i32 @main() {
mainEntry:
  %a = alloca i32, align 4
  store i32 2, i32* %a, align 4
  br label %condition

condition:                                        ; preds = %mainEntry
  %a1 = load i32, i32* %a, align 4
  %lt = icmp slt i32 %a1, 1
  %cond = zext i1 %lt to i32
  %lt2 = icmp slt i32 %cond, 2
  %cond3 = zext i1 %lt2 to i32
  %cond4 = icmp ne i32 0, %cond3
  br i1 %cond4, label %the_true, label %the_false

the_true:                                         ; preds = %condition
  store i32 1, i32* %a, align 4
  br label %the_next

the_false:                                        ; preds = %condition
  br label %the_next

the_next:                                         ; preds = %the_false, %the_true
}
