; ModuleID = 'moudle'
source_filename = "moudle"

@a = global i32 10

define i32 @main() {
mainEntry:
  br label %condition

condition:                                        ; preds = %mainEntry
  %a = load i32, i32* @a, align 4
  %ne = icmp ne i32 %a, 10
  %cond = zext i1 %ne to i32
  %cond1 = icmp ne i32 0, %cond
  br i1 %cond1, label %the_true, label %the_false

the_true:                                         ; preds = %condition
  store i32 2, i32* @a, align 4
  br label %the_next

the_false:                                        ; preds = %condition
  store i32 20, i32* @a, align 4
  br label %the_next

the_next:                                         ; preds = %the_false, %the_true
  %a2 = load i32, i32* @a, align 4
  ret i32 %a2
}
